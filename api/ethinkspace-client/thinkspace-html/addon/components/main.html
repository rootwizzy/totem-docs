import ember from 'ember' import base from 'thinkspace-base/components/base' import prerp from 'thinkspace-html/preprocessors/responses' import precf from 'thinkspace-html/preprocessors/carry_forward' export default base.extend totem_data_config: ability: {ajax_source: true} tvo_titles: ['html', 'html-select-text', 'html-only'] preprocess: ['html', 'html_select_text'] html_content: null init_base: -> @preprocess_html_content().then (html_content) => return if @is_destroyed() # ensure current user did not switch to another user (e.g. via gradebook) before prepocessing was completed @set('html_content', html_content) @set_all_data_loaded() preprocess_html_content: -> new ember.RSVP.Promise (resolve, reject) => return resolve() if @is_destroyed() model = @get('model') tvo = @get('tvo') html_content = model.get('html_content') html_content = html_content.replace(/<\/input>/g, '') # fix content html_content = html_content.replace(/<\/br>/g, '<br>') # fix content return resolve(html_content) unless @preprocess.includes(@component_name) responses = prerp.create(tvo: tvo, tc: @tc, input_element_root_path: 'input_elements', input_element_component_prefix: 'standard' ) carry_forward = precf.create(tvo: tvo, tc: @tc, input_element_root_path: 'input_elements', input_element_component_prefix: 'standard' ) responses.process(model, html_content).then (html_content) => carry_forward.process(model, html_content).then (html_content) => resolve(html_content)
