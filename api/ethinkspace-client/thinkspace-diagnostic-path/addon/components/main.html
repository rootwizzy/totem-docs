import ember from 'ember' import util from 'totem/util' import base from 'thinkspace-base/components/base' import source_observation from 'thinkspace-diagnostic-path/sources/observation' import source_mechanism from 'thinkspace-diagnostic-path/sources/mechanism' export default base.extend tvo_titles: 'diagnostic-path' init_base: -> tvo = @get('tvo') hash = tvo.template.dup_value(@tvo_path) hash.title = 'indented-list' hash.indented_list = @get_indented_list_config() @indented_path = tvo.value.set_value(hash) @tvo_status_register_callback(@, 'validate_diagnostic_path') @register_remove_itemable() @set_all_data_loaded() get_indented_list_config: -> hash = {} hash.sources = [source_observation.create(), source_mechanism.create()] hash.indent_px = 20 hash

#

Helpers.

#

get_source_section: -> @get('attributes.source') # section-name in the template's <component ... source='section-name'/> get_response_manager: -> @get('tvo').get_path_value("#{@indented_path}.response_manager") get_response_manager_items: -> @get_response_manager().value_items

#

Remove Itemable from Path.

#

When an itemable component destroys an itemable (e.g. observation-list observation), the component first calls the registered method (which removes from the path list).

register_remove_itemable: -> section = @get_source_section() return if ember.isBlank(section) # no source itemables, so don't need to register @tvo_section_register_actions section: 'remove_itemable' actions: remove: 'remove_itemable_in_items' remove_itemable_in_items: (itemable) -> new ember.RSVP.Promise (resolve, reject) => @get_response_manager().remove_items_with_itemable(itemable).then => resolve()

#

Validate Path (all itemables are in path).

#

validate_diagnostic_path: -> new ember.RSVP.Promise (resolve, reject) => tvo = @get('tvo') messages = [] action = 'itemables' section = @get_source_section() return resolve() unless section # return if no source (e.g. an observation list) return resolve() unless @tvo_section_has_action(section, action) # did not register an 'itemables' action @tvo_section_call_action(section, action).then (itemables) => return resolve("No observations present.") if ember.isBlank(itemables) response = @get('model.responses.firstObject') return resolve('Response is blank.') if ember.isBlank(response) rm = @get_response_manager() return resolve('Response Manager is blank.') if ember.isBlank(rm) items = @get_response_manager_items() return resolve("The diagnostic path is blank.") if ember.isPresent(itemables) and ember.isBlank(items) itemables.forEach (itemable) => messages.push @get_itemable_unused_message(itemable) unless rm.is_itemable_used_in_items(itemable) return resolve() if ember.isBlank(messages) count = messages.length title = "All observations must be included in the diagnostic path. Unused #{util.pluralize('observation', count)}:" resolve tvo.status.message_ordered_list(title, messages) get_itemable_unused_message: (itemable) -> itemable.get('value')
