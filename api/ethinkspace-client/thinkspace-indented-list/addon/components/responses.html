import ember from 'ember' import ns from 'totem/ns' import ta from 'totem/ds/associations' import base from 'thinkspace-base/components/base' export default base.extend classNames: ['indented-list'] responses: ember.computed -> promise = new ember.RSVP.Promise (resolve, reject) => sort_by = 'ownerable_id' @get_ownerable_responses().then (responses) => if ember.isBlank(responses) resolve [@create_response()] else resolve(responses.sortBy sort_by) ta.PromiseArray.create promise: promise create_response: -> store = @totem_scope.get_store() list = @get('model') response = @tc.create_record ns.to_p('indented:response'), ownerable_id: @totem_scope.get_ownerable_id() ownerable_type: @totem_scope.get_ownerable_type() value: items: [] response.set ns.to_p('indented:list'), list response get_ownerable_responses: -> new ember.RSVP.Promise (resolve, reject) => tvo = @get('tvo') responses = tvo.get_path_value("#{@tvo_path}.indented_list.responses") return resolve(responses) if responses list = @get('model') @tc.view_load(list).then => resolve list.get('responses')
