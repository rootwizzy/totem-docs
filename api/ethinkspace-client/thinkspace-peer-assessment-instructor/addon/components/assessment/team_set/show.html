import ember from 'ember' import ns from 'totem/ns' import base_component from 'thinkspace-base/components/base' export default base_component.extend

Intialization

init_base: -> @init_team_set().then => @init_team().then => @init_review_sets().then => @init_ignored_ownerables() @set_all_data_loaded() init_team_set: -> new ember.RSVP.Promise (resolve, reject) => @tc.find_record(ns.to_p('tbl:team_set'), @get('team_set_data.id')).then (team_set) => @set 'model', team_set resolve() init_team: -> new ember.RSVP.Promise (resolve, reject) => model = @get('model') model.get('team').then (team) => @set 'team', team @set 'team_members', team.get('users') resolve() init_review_sets: -> new ember.RSVP.Promise (resolve, reject) => model = @get('model') model.get(ns.to_p('tbl:review_sets')).then (review_sets) => @set 'review_sets', review_sets resolve() init_ignored_ownerables: -> review_sets = @get('review_sets').filterBy 'is_not_complete' ownerable_ids = review_sets.mapBy 'ownerable_id' ignored_ownerables = @get('team_members').filter (ownerable) => ownerable_ids.contains(parseInt(ownerable.get('id'))) @set 'ignored_ownerables', ignored_ownerables

Helpers

state_change: (state) -> model = @get 'model' query = id: model.get('id') options = action: state verb: 'PUT' @tc.query_action(ns.to_p('tbl:team_set'), query, options).then (review_set) => @update_progress_report(review_set) update_progress_report: (review_set) => progress_report = @get_progress_report() team_set = @get 'team_set' progress_report.process_review_sets(team_set, review_set) get_progress_report: -> @get 'progress_report'

Actions

actions: toggle_approve: -> if @get('model.is_approved') then @state_change('unapprove') else @state_change('approve') select_team_set: -> @sendAction 'select_team_set', null
