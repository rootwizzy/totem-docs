position_from_value: -> position = @get 'model.value.position' height = @get 'height' width = @get 'width' height_offset = parseFloat (height / 2.0) width_offset = parseFloat (width / 2.0) render_x = position.x - width_offset y = position.y page = position.page page_top = $("canvas[page='#{page}'").position().top # Page height is off by 4px for each page except first. render_y = if page > 1 then (y + page_top - height_offset) else (y - height_offset) @$().css top: render_y left: render_x
