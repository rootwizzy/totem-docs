import ember from 'ember' import Resolver from 'ember-engines/resolver' export default Resolver.extend resolveTemplate: (pn) -> @set_name(pn) @_super(pn) resolveOther: (pn) -> if pn.type == 'helper' @set_helper(pn) else @underscore_pn(pn) @set_name(pn) @_super(pn) set_helper: (pn) -> return unless pn.name.match('--') @remove_helper_identifier(pn) @set_app_as_owner(pn) set_name: (pn) -> return unless pn.name.match('__') @remove_app_identifier(pn) @set_app_as_owner(pn) set_app_as_owner: (pn) -> engine = @get_owner(pn.root) router = engine.lookup('router:main') owner = @get_owner(router) mod_prefix = owner.application.modulePrefix pn.root = owner pn.prefix = mod_prefix get_owner: (root) -> Ember.getOwner(root) remove_app_identifier: (pn) -> pn[prop] = pn[prop].replace('__', '') for prop in @get_name_variables() remove_helper_identifier: (pn) -> pn[prop] = pn[prop].replace('--', '') for prop in @get_name_variables() underscore_pn: (pn) -> pn[prop] = @underscore_string(pn[prop]) for prop in @get_name_variables() get_name_variables: -> ['fullName', 'fullNameWithoutType', 'name'] underscore_string: (val) -> (val and ember.String.underscore(val)) or ''
