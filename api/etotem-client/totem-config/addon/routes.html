import ember from 'ember' import util from 'totem/util_ember' import tc from 'totem-config/configs' class TotemRoutes constructor: -> @engine_mounts = {} @engine_mount_as = [] @engine_mount_paths = [] @populate_routes() populate_routes: -> mounts = tc.get_router_mounts() return if ember.isBlank(mounts) @add_mount(mount) for mount in mounts add_mount: (mount) -> @error "Mount is not a hash.", mount unless util.is_hash(mount) mod = mount.module engine = mount.engine args = mount.args @error "Module '#{mod}' engine mount name is not a string.", mount unless util.is_string(engine) @error "Module '#{mod}' engine mount args is not a hash.", mount unless util.is_hash(args) @error "Module '#{mod}' engine mount is a duplicate.", mount if ember.isPresent(@engine_mounts[engine]) as = args.as @error "Module '#{mod}' engine mount args 'as' key is blank.", mount if ember.isBlank(as) @error "Module '#{mod}' engine mount as '#{as}' is a duplicate.", mount if @engine_mount_as.includes(as) path = args.path or as @error "Module '#{mod}' engine mount path '#{path}' is a duplicate.", mount if @engine_mount_paths.includes(path) @engine_mounts[engine] = args @engine_mount_as.push(as) # for dup check @engine_mount_paths.push(path) # for dup chech
