require 'model_helper' module Test; module WeatherForecaster; module Model; class ResponseTest < ActiveSupport::TestCase include ModelHelper include ResponseHelper describe 'calculate score' do include ResponseDefaultLets describe 'default values' do it 'is correct' do; assert_equal default_score, get_score; end it 'no options' do; assert_equal default_score, get_score_by_options; end it 'actual in options' do; assert_equal default_score, get_score_by_options(forecast_day_actual: forecast_day_actual); end end describe 'is correct' do describe 'has item values' do let(:correct) {5} it 'is correct' do; assert_equal 5, get_score; end end describe 'has assessment item values' do let(:ai_correct) {6} it 'is correct' do; assert_equal 6, get_score; end end end describe 'no item processing' do let(:correct) {nil}; let(:incorrect) {nil} it 'score is zero' do; assert_equal 0, get_score; end end describe 'not correct' do describe 'is default score' do let(:input_value) {'x'}; let(:incorrect) {7} it 'is correct' do; assert_equal 7, get_score; end end end end # calc score describe 'response score metadata' do include ResponseDefaultLets describe 'metadata is nil' do it 'nil' do; assert_nil response.response_score_metadata, 'does not have a response score record'; end end describe 'input is correct' do let(:correct) {5} it 'metadata' do expect = {is_correct: true, var_actual: actual_val, score: 5} assert_equal expect, get_response_score_metadata end end describe 'input is not correct' do let(:incorrect) {5} let(:input_value) {'2'} it 'metadata' do expect = {is_correct: false, var_actual: actual_val, score: 5} assert_equal expect, get_response_score_metadata end end end # response score metadata end; end; end; end
