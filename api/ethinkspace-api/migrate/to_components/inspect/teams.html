'---NO TEAMABLE---' : "#{teamable.teamable_type}.#{teamable.teamable_id}" puts message no_teamables.push(team) if teamable.blank? && team.authable_type.blank? end puts "\n" if no_teamables.present? puts "NO TEAMABLES:" no_teamables.each do |team| puts "team id: #{team.id.to_s.rjust(4)} authable: #{team.authable_type.inspect.ljust(30)} title: #{team.title.inspect.ljust(30)}" end else puts "All teams have an authable or teamable" end end def old_team_class; get_old_model_class('thinkspace/team/team'); end def old_team_teamable_class; get_old_model_class('thinkspace/team/team_teamable'); end end end; end; end; end
